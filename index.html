📌 Struktur Firestore Database
Collection: transaksi
  Document: {auto-id}
    tanggal   : "2026-03-03"    ← string YYYY-MM-DD
    tipeKas   : "in" | "out"
    deskripsi : "Penjualan Produk A"
    kategori  : "Penjualan"
    jenis     : "Tunai"
    tipe      : "Penjualan Produk"
    nominal   : 2500000          ← number
    catatan   : ""
    createdAt : Timestamp        ← serverTimestamp()
    updatedAt : Timestamp

Collection: config
  kategori → { items: [{id, nama, warna}] }
  jenis    → { items: ["Tunai", "Transfer",...] }
  tipeIn   → { items: ["Penjualan Produk",...] }
  tipeOut  → { items: ["Pembelian Bahan",...] }
🔒 Firestore Security Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /transaksi/{docId} {
      allow read, write: if request.auth != null;
    }
    match /config/{docId} {
      allow read, write: if request.auth != null;
    }
  }
}
⚡ Query Rekap Harian (JavaScript SDK)
// Query transaksi per bulan
const q = query(
  collection(db, "transaksi"),
  where("tanggal", ">=", "2026-03-01"),
  where("tanggal", "<=", "2026-03-31"),
  orderBy("tanggal", "desc")
);

// Query transaksi per hari tertentu
const qDay = query(
  collection(db, "transaksi"),
  where("tanggal", "==", "2026-03-03")
);
